1) 환경 설정: application.yml "라인별 왜?"

코드 (한 줄마다 이유 주석) 
spring: # 스프링 설정의 최상위 묶음입니다. 여기 아래에 DB/JPA/템플릿 설정을 넣습니다.
  datasource: # DB 연결 정보 묶음입니다.
    url: jdbc:postgresql://localhost:5432/spring_lv1 # jdbc:postgresql(프로토콜)://localhost(내 컴퓨터):5432(DB 기본 포트)/spring_lv1(DB 이름)
    username: postgres # DB 계정명입니다. DB 문을 열 때 필요한 아이디입니다.
    password: 1234 # DB 비밀번호입니다. 학습 단계에서는 단순값, 실무는 환경변수 사용.
    driver-class-name: org.postgresql.Driver # 자바가 PostgreSQL과 대화하기 위한 드라이버 클래스입니다.

  jpa: # 자바 객체(Entity)와 DB 테이블을 연결하는 JPA 설정입니다.
    hibernate: # JPA 구현체(Hibernate) 상세 설정 구역입니다.
      ddl-auto: update # Entity가 바뀌면 테이블을 업데이트합니다. 학습 속도를 높여줍니다.
    show-sql: true # 실제 실행 SQL을 콘솔에 보여줍니다. 내부 동작 학습에 필수입니다.
    properties: # 추가 하이버네이트 옵션 묶음입니다.
      hibernate:
        format_sql: true # SQL을 예쁘게 줄바꿈해서 출력합니다. 초보자 가독성 향상.

  thymeleaf: # HTML 템플릿 엔진 설정입니다.
    cache: false # 화면 파일 수정 즉시 반영. 학습 중에는 캐시 OFF가 편합니다.

server: # 내장 톰캣 서버 설정입니다.
  port: 8080 # 브라우저 접속 포트(예: http://localhost:8080)
문법 해설 (아주 기초)
  • YAML은 키: 값 구조입니다. 들여쓰기 깊이가 같으면 같은 레벨입니다.
  • localhost는 "내 컴퓨터 자신"을 뜻합니다.
  • 5432는 PostgreSQL 기본 포트입니다. 바뀌면 접속 주소도 같이 바꿔야 합니다.
  • ddl-auto: update는 편하지만, 운영에서는 의도치 않은 변경 위험이 있어 주의가 필요합니다.

2) 데이터 설계: Member Entity를 왜 이렇게 만드는가?

필드 선정 사고방식
  • id: 사람 이름은 중복될 수 있으므로 시스템 내부 식별자는 숫자 PK가 안전합니다.
  • email: 로그인 아이디 역할. 절대 중복되면 안 됩니다.
  • password: 인증 핵심. 저장은 필요하지만 평문 저장은 금지입니다.
Member 코드 (라인별 이유 주석) 
package com.example.lv1.entity; // 이 클래스가 entity 폴더에 있음을 알려줍니다.

import jakarta.persistence.Column; // 컬럼 옵션(nullable, unique, length)을 쓰기 위해 필요합니다.
import jakarta.persistence.Entity; // 이 클래스가 DB 테이블로 매핑된다는 표시입니다.
import jakarta.persistence.GeneratedValue; // id 자동 생성 기능을 쓰기 위해 필요합니다.
import jakarta.persistence.GenerationType; // 자동 생성 방식 종류를 지정하기 위해 필요합니다.
import jakarta.persistence.Id; // 기본키 필드 표시를 위해 필요합니다.
import lombok.Builder; // 빌더 패턴 사용을 위해 필요합니다.
import lombok.Getter; // getter 자동 생성을 위해 필요합니다.
import lombok.NoArgsConstructor; // 기본 생성자 자동 생성을 위해 필요합니다.

@Entity // JPA가 이 클래스를 테이블로 관리하게 만듭니다.
@Getter // private 필드를 외부에서 읽을 수 있도록 getter를 자동 생성합니다.
@NoArgsConstructor // JPA 내부 동작 시 기본 생성자가 필요해서 붙입니다.
public class Member { // Member 엔티티 클래스 선언

    @Id // PK(기본키) 지정
    @GeneratedValue(strategy = GenerationType.IDENTITY) // DB auto increment 전략 사용
    private Long id; // 회원 고유 번호. Long은 큰 숫자 범위와 null 상태 표현에 유리

    @Column(nullable = false, unique = true, length = 100) // null 금지 + 중복 금지 + 최대 100자
    private String email; // 로그인 아이디. unique를 반드시 주는 대표 필드

    @Column(nullable = false, length = 255) // null 금지 + 최대 255자
    private String password; // 암호화된 비밀번호 저장 칸

    @Builder // Member.builder() 형태 생성 문법 제공
    public Member(String email, String password) { // 생성 시 반드시 필요한 필드만 받도록 설계
        this.email = email; // 전달받은 email을 현재 객체 필드에 대입
        this.password = password; // 전달받은 password를 현재 객체 필드에 대입
    }
}
Java 기초 연결 (private, Long, String)
요소 뜻 왜 여기서 선택했나?
private 외부 직접 접근 금지 필드 보호. Service 규칙을 거치게 만들기 위해
Long 큰 정수 객체형 PK 범위가 넓고, 저장 전 null 상태 표현 가능
String 문자열 타입 email/password는 문자 데이터이기 때문
this 현재 객체 자신 매개변수명과 필드명을 명확히 구분
Builder 패턴 상세 분석 
Member member = Member.builder() // "객체 만들기 시작" 신호
    .email("student@test.com") // 어떤 필드에 어떤 값을 넣는지 글자 그대로 보임
    .password("암호화된비밀번호") // 생성자 순서 실수 방지
    .build(); // 최종 객체 생성
  • 장점 1: 가독성. 어떤 값이 어떤 필드로 가는지 코드가 말해줍니다.
  • 장점 2: 순서 실수 방지. new Member(a, b)보다 안전합니다.
  • 장점 3: 필드가 늘어나도 읽기/수정이 쉽습니다.

3) Controller 기초: public String joinForm(Model model) 해부

코드 (라인별 주석) 
package com.example.lv1.controller; // 컨트롤러 패키지

import com.example.lv1.dto.MemberJoinRequest; // 화면 폼 데이터 바인딩 DTO
import org.springframework.stereotype.Controller; // MVC 컨트롤러 선언용
import org.springframework.ui.Model; // 컨트롤러 -> 뷰 데이터 전달 상자
import org.springframework.web.bind.annotation.GetMapping; // GET 요청 매핑
import org.springframework.web.bind.annotation.RequestMapping; // 공통 경로 매핑

@Controller // 이 클래스가 HTML 화면을 반환하는 컨트롤러임을 뜻합니다.
@RequestMapping("/members") // 클래스 내부 메서드 URL 공통 접두사
public class MemberController {

    @GetMapping("/join") // GET /members/join 요청을 이 메서드와 연결
    public String joinForm(Model model) { // public: 스프링이 호출 가능, String: 반환값은 뷰 이름, Model: 화면으로 보낼 데이터 통로
        model.addAttribute("req", new MemberJoinRequest()); // req라는 이름으로 빈 DTO 전달
        return "member/join"; // templates/member/join.html 렌더링
    }
}
문법 부품 분해표
부품 문법 종류 왜 필요한가?
public 접근 제어자 스프링 프레임워크가 메서드를 호출하려면 공개되어야 함
String 반환형 어떤 HTML 템플릿을 보여줄지 뷰 이름으로 반환
joinForm 메서드 이름 역할(회원가입 폼 화면 제공)을 이름에서 바로 알 수 있게
Model model 파라미터 컨트롤러에서 뷰로 데이터를 넘기기 위한 운반 상자
model.addAttribute 메서드 호출 템플릿이 사용할 데이터를 이름(req)과 함께 등록
return "member/join" 반환문 최종적으로 어떤 화면 파일을 보여줄지 결정

4) 어노테이션 사전 (Lv.1 사용분)

동작 원리 + 사용 시점
  • @Entity: 클래스가 DB 테이블 매핑 대상임을 선언. 데이터 설계 단계에서 사용.
  • @Column: 컬럼 제약(필수/중복/길이) 지정. 무결성 보호를 위해 사용.
  • @Id, @GeneratedValue: PK 정의 및 생성 전략 설정.
  • @Getter, @NoArgsConstructor, @Builder: Lombok으로 반복 코드 제거 + 객체 생성 가독성 향상.
  • @Controller, @RequestMapping, @GetMapping: URL 요청을 화면 반환 메서드와 연결.
